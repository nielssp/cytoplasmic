    Returns MutCell<T2>

    Example

    const a = bind(1);
    const b = a.bimap(x => x + 1, y => y - 1);

    expect(b.value).toBe(2);

    a.value = 5;
    expect(b.value).toBe(6);

    b.value = 3;
    expect(a.value).toBe(2);
